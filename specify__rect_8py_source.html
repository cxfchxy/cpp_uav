336  # Create a list of vertices

339  waypoint_ys = []

361  for num, point in enumerate(self.points["waypoints"]):

364  waypoint_ys.append(self.points["start"].y)
365  self.labels["SP"] = self.axis.text(
366  point.x, point.y, "SP", color="red", fontsize=16)
367  waypoint_xs.append(point.x)
368  waypoint_ys.append(point.y)
369  if num == len(self.points["waypoints"]) - 1:
370  waypoint_xs.append(self.points["start"].x)
371  waypoint_ys.append(self.points["start"].y)
372  self.labels["EP"] = self.axis.text(
373  point.x, point.y, "EP", color="red", fontsize=16)
